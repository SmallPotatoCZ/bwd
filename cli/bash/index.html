for

语法

# words 不指定，只执行一次
for name [in words ...]; do commands; done

for (( expr1; expr2 )); do commands; done # 等价于
((  expr1 ))
while (( expr2 ));do
    commands
done

words

file1.txt file2.txt file3.txt
# 或者
Apple Sony "Hewlett Packard"

通配符 *

表示的是文件夹中的文件名字

for thisfile in *; do
    echo "found $thisfile";
done

例子

1 - 10

for i in `seq 1 10`
do
    echo "$i";
done
# 大括号实现
for i in {1..10}
do
    echo "$i";
done
# 一行的表示
for i in {1..10}; do echo $i; done

for i in $(seq 1 100); do echo -n "Hello World${i} "; done

test-command

语法

# [[]] 为扩展的 test，推荐使用
test expr
    [ expr ]
    [[ expr ]]

使用变量，你应该用"variablename"的格式，在[[]]中是不必须的。

[ "$DEMO" == 5 ];
[[ $DEMO == 10 ]];

多个表达式

!

表达式为false时返回true

! exp

()

返回表达式的值。

( expr )

&&

表达式 expr1 和 expr2 都为true

expr1 && expr2

||

表达式 expr1 或 expr2 的值为true

expr1 || expr2

文件类型

-d file # 文件时一个文件夹
-f file # 文件是一个标准的文件

字符类型

-z string # 字符的长度为 0
-n string # 字符的长度不为 0
    string # 字符的长度不为 0
striing1 = string2 # 字符相等
string1 != string2 # 两字符不相等
[[ a != b ]] # a 不等于 b
[[ a > b ]] # a 大于 b
[[ a < b ]] # a 小于 b
[[ a = a ]] # a 等于 a

数字类型

数字的比较应该用 (( ))或者最新的[[ ]]

ARG1 -eq ARG2 # ARG1 等于 ARG2
ARG1 -ne ARG2 # ARG1 不等于 ARG2
ARG1 -lt ARG2 # ARG1 小于 ARG2
ARG1 -le ARG2 # ARG1 小于等于 ARG2
ARG1 -gt ARG2 # ARG1 大于 ARG2
ARG1 -ge ARG2 # ARG1 大于等于 ARG2
